### 代码

- 使用`toString()`进制转化函数
```js
/**
 * @param {string} date
 * @return {string}
 */
var convertDateToBinary = function (date) {
  return date.split('-').map(ele => Number(ele).toString(2)).join('-');
};
```

- 二进制算法
```js
var convertDateToBinary = function (date) {
  return date.split('-').map(ele => ele.toBinary()).join('-');
};

String.prototype.toBinary = function () {
  let num = +this;
  let str = '';
  while (num > 0) {
    str = num % 2 + str;
    num = Math.floor(num / 2);
  }
  return str;
}
```
