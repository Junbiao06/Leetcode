/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} x
 * @return {ListNode}
 */
var partition = function (head, x) {
  let headA = headB = tailA = tailB = null;
  while (head) {
    let next = head.next;
    head.next = null;
    if (head.val < x) {
      if (headA === null) {
        headA = tailA = head;
      }
      else {
        tailA.next = head;
        tailA = tailA.next
      }
    }
    else {
      if (headB === null) {
        headB = tailB = head;
      }
      else {
        tailB.next = head;
        tailB = tailB.next;
      }
    }
    head = next;
  }
  if (!headA) return headB
  if (!headB) {
    return headA;
  }
  tailA.next = headB;
  return headA;
};
