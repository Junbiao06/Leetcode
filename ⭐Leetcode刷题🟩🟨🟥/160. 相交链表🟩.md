### 代码
平台方法，遍历两遍，更巧妙
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} headA
 * @param {ListNode} headB
 * @return {ListNode}
 */
var getIntersectionNode = function (head1, head2) {
  let p1 = head1;
  let p2 = head2;
  while (p1 !== p2) {
    p1 = p1 ? p1.next : head2;
    p2 = p2 ? p2.next : head1;
  }
  return p1;
};
```

左神课堂，笨方法
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} headA
 * @param {ListNode} headB
 * @return {ListNode}
 */
var getIntersectionNode = function (head1, head2) {
  if (!head1 || !head2) return null;
  let cur1 = head1;
  let cur2 = head2;
  let n = 0
  while (cur1.next) {
    n++
    cur1 = cur1.next;
  }
  while (cur2.next) {
    n--
    cur2 = cur2.next;
  }
  if (cur1 !== cur2) return null;
  cur1 = n > 0 ? head1 : head2;
  cur2 = cur1 === head2 ? head1 : head2;
  n = Math.abs(n);
  while (n != 0) {
    n--;
    cur1 = cur1.next;
  }
  while (cur1 !== cur2) {
    cur1 = cur1.next;
    cur2 = cur2.next;
  }
  return cur1;
};
```
