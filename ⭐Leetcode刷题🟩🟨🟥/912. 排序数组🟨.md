/**
 * @param {number[]} nums
 * @return {number[]}

// mergeSort
var sortArray = function (nums) {
  if (nums === null || nums.length < 2) return nums;
  merge(nums, 0, nums.length - 1)
  return nums;
};
const merge = function (nums, l, r) {
  if (l === r) return nums[l];
  const m = Math.floor(l + (r - l) / 2);
  merge(nums, l, m);
  merge(nums, m + 1, r);
  mergeTwoParts(nums, l, m, r);
}
const mergeTwoParts = function (nums, l, m, r) {
  const help = []
  let i = 0
  p1 = l
  p2 = m + 1
  while (p1 <= m && p2 <= r) {
    if (nums[p1] <= nums[p2])
      help[i++] = nums[p1++]
    else {
      help[i++] = nums[p2++]
    }
  }
  while (p2 <= r) help[i++] = nums[p2++]
  while (p1 <= m) help[i++] = nums[p1++]
  for (let i = 0; i < help.length; i++) {
    nums[l + i] = help[i];
  }
}
