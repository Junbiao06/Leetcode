/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var rotateRight = function (head, k) {
  if (!head || !head.next) return head;
  let p = head;
  let length = 1;
  while (p.next) {
    length++;
    p = p.next;
  }
  p.next = head;
  for (let i = 1; i <= length - (k % length); i++) {
    if (i === length - (k % length)) {
      let h = head.next;
      head.next = null;
      return h;
    }
    head = head.next;
  }
};

/**

 */
var rotateRight = function (head, k) {
  let p = head
  let length = 0;
  if (!p || !p.next || k === 0) return head
  if (!p.next.next) {
    if (k % 2 === 0) return head;
    else {
      let h = head.next
      head.next.next = head;
      head.next = null;
      head = h;
      return head;
    }
  }
  p = head
  while (p) {
    p = p.next
    length++;
  }
  for (let i = 0; i < k % length; i++) {
    p = head
    while (p.next.next) {
      p = p.next;
    }
    let h = p.next;
    p.next = null;
    h.next = head;
    head = h;
  }
  return head;

};
